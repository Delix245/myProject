# Добавляем все файлы с расширением .txt в текущей директории

git add *.txt


# Добавляем все .txt файлы в подпапках (один уровень)

git add */*.txt


# Показываем текущий статус: какие файлы идут на коммит

git status


# Удалить файл из индекса (ожидания), но оставить его на диске

git rm --cached <file>


# Если Git требует "use -f to force removal", добавляем -f:

git rm --cached -f <file>


# Удалить все файлы из индекса (не с диска)


git rm --cached -r -f .



# чтоб выгрузить на сервер локальный

git commit


# чтоб добавить сообщение когда например изменил файл


git commit -m "first version of project"

# Створення файлу example.txt
touch example.txt


# удаляем файл из подготовки
.git restore --staged .gitignore
